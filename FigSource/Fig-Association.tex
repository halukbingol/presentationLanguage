\newcommand{\xyAxis}[6]{% \xyAxis ==== V
	\def\xMax{#1}
	\def\yMax{#2}
	\def\xValue{#3}
	\def\yValue{#4}
	\def\xName{#5}
	\def\yName{#6}

	\draw[gray, very thin] 
		(0, 0) grid (\xMax, \yMax); 
		
	% x-axis
	\draw [->,thick, orange]
		(0, 0) -- ({(\xMax+0.5)}, 0)
		node[anchor=north, xshift=-0 cm] {\xName};
	\foreach \i in {1, ..., \xMax}{
		\draw 
			(\i, -0.05) -- (\i, 0.05)
			node[anchor=north, xshift=-0 cm] {$\xValue_{\i}$};
	}
			
	% y-axis
	\draw [->,thick, orange]
		(0, 0) -- (0, {(\yMax+0.5)})
		node[anchor=east, xshift=-0 cm] {\yName};
	\foreach \i in {1, ..., \yMax}{
		\draw 
			(-0.05, \i) -- (0.05, \i)
			node[anchor=east, xshift=-0 cm] {$\yValue_{\i}$};
	}
} % \xyAxis ==== A
